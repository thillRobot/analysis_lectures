\subsection{Part 2 - Euler's Method}
\frame[containsverbatim]{
  \frametitle{Part 2 - Euler's Method}

% This method is not suitable for manual computation. \vspace{0mm}\\

 \begin{lstlisting}
% approximate with Euler's forward integration
z1_eu(1)=theta0*pi/180; z2_eu(1)=0;  % initial conditions

for j=1:length(time)-1
    z1_eu(j+1)=z1_eu(j)+f1(time(j),z1_eu(j),z2_eu(j))*dt;  
    z2_eu(j+1)=z2_eu(j)+f2(time(j),z1_eu(j),z2_eu(j))*dt;
end

function [dz1dt]=f1(t,z1,z2)
    global m g l kt;
    dz1dt=z2;
end
function [dz2dt]=f2(t,z1,z2)
    global m g l kt;
    dz2dt=(m*g*l*sin(z1)-kt*z1)/(m*l^2);
end
  \end{lstlisting}

}
