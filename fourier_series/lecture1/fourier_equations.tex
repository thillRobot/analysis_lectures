\textbf{ \LARGE ME 3023 - The Fourier Series} \\\\

\begin{itemize}

	\item  \textbf{\LARGE The Fourier Series - Full Expansion} \vspace{10mm}\\

	\scalebox{2}{$f(x)=a_0 + \Sigma_{n=1}^{\infty} \{a_n cos(nx)+b_n sin(nx) \}$}\vspace{5mm}\\

	\scalebox{2}{$a_0=\frac{1}{2\pi}\int_{-\pi}^{\pi}f(x)dx$} \vspace{5mm}\\

	\scalebox{2}{$a_n=\frac{1}{\pi}\int_{-\pi}^{\pi}f(x)cos(nx)dx$} \vspace{5mm}\\
	
	\scalebox{2}{$b_n=\frac{1}{\pi}\int_{-\pi}^{\pi}f(x)sin(nx)dx$} \vspace{5mm}\\
	

	\item \textbf{\LARGE Alternate Form (more useful)} \vspace{10mm}\\
	
	\scalebox{2}{$f(x)=a_0 + \Sigma_{n=1}^{\infty} \{a_n cos(\frac{n\pi x}{P})+b_n sin(\frac{n\pi x}{P}) \}$} \vspace{5mm}\\

	\scalebox{2}{$a_0=\frac{1}{2P}\int_{-P}^{P}f(x)dx$} \vspace{5mm}\\

	\scalebox{2}{$a_n=\frac{1}{P}\int_{-P}^{P}f(x)cos(\frac{n\pi x}{P})dx$} \vspace{5mm}\\
	
	\scalebox{2}{$b_n=\frac{1}{P}\int_{-P}^{P}f(x)sin(\frac{n\pi x}{P})dx$} \vspace{5mm}\\
	
	 \vspace{20mm}
	
	
	
	
	
	
	\newpage
	
	\item  \textbf{\LARGE The Fourier Series - Half Range Expansions} \vspace{10mm}\\
	
	\begin{itemize}
		\item \textbf{\Large Cosine Series } \vspace{5mm}\\
		
			\scalebox{2}{$f(x)=a_0 + \Sigma_{n=1}^{\infty} \{a_n cos(\frac{n\pi x}{P})\}$} \vspace{10mm}\\

			\scalebox{2}{$a_0=\frac{1}{P}\int_{0}^{P}f(x)dx$} \vspace{10mm}\\

			\scalebox{2}{$a_n=\frac{2}{P}\int_{0}^{P}f(x)cos(\frac{n\pi x}{P})dx$} \vspace{10mm}\\
		
		\item \textbf{\Large Sine Series } \vspace{5mm}\\
		
			\scalebox{2}{$f(x)=\Sigma_{n=1}^{\infty} \{b_n sin(\frac{n\pi x}{P})\}$} \vspace{10mm}\\

			\scalebox{2}{$b_n=\frac{2}{P}\int_{0}^{P}f(x)sin(\frac{n\pi x}{P})dx$} \vspace{5mm}\\
		
\end{itemize}
	\item  \textbf{\LARGE Some Useful Stuff...} \vspace{10mm}\\
	
		\scalebox{2}{$sin(n\pi)=0$ \hspace{5mm} if $n$ is an integer} \vspace{3mm}\\
		\scalebox{2}{$cos(n\pi)=(-1)^n$ \hspace{5mm} if $n$ is an integer}


\end{itemize}
